44  template <typename I, typename S>
45  S join(I first, I last, S const &separator, S result) {
46  if (first == last) {
47  return result;
48  }
49  I prev = first;
50  I next = first;
51  for (++next; next != last; ++next) {
52  result += *prev;
53  result += separator;
54  prev = next;
55  }
56  result += *prev;
57  return result;
58  }
